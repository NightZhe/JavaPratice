-- CREATE TABLE
CREATE TABLE STUDENT( 
	SNO VARCHAR(10) PRIMARY KEY, 
	SNAME VARCHAR(20), 
	SBDAY DATETIME, 
	SSEX INT,
	SMAIL VARCHAR(50),
	SPWD VARCHAR(32)
);

CREATE TABLE TEACHER( 
	TNO VARCHAR(10) PRIMARY KEY, 
	TNAME VARCHAR(20),
	TBDAY DATETIME, 
	TSEX INT,
	TMAIL VARCHAR(50),
	TPWD VARCHAR(32)
);

CREATE TABLE COURSE( 
	CNO VARCHAR(10), 
	CNAME VARCHAR(20), 
	TNO VARCHAR(20), 
	PRIMARY KEY (CNO,TNO) 
);

CREATE TABLE SC( 
	SNO VARCHAR(10), 
	CNO VARCHAR(10), 
	SCORE INT, 
	PRIMARY KEY (SNO,CNO) 
);

-- DROP TABLE
DROP TABLE STUDENT;
DROP TABLE TEACHER;
DROP TABLE COURSE;
DROP TABLE SC
